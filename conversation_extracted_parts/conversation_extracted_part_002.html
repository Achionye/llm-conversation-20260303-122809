from __future__ import annotations
from pathlib import Path
import csv, re, os

ROOT = Path(__file__).resolve().parent.parent
REG  = ROOT / "screening_v1" / "study_registry_SCREENING_WORKING.csv"
PDFDIR = ROOT / "forlitrev_pdfs"
TEIDIR = ROOT / "fulltext_tei_from_pdfs_v1"
OUT = ROOT / "out_synthesis_filtered_full_v1" / "included_primaryfile_to_tei_map_v1.csv"

def clean(s): return re.sub(r"\s+"," ",(s or "")).strip()

def resolve_pdf_path(primary_file: str) -> Path | None:
    if not primary_file:
        return None
    p = Path(primary_file)

    # absolute path
    if p.is_absolute():
        return p if p.exists() else None

    # relative to project root
    p1 = ROOT / p
    if p1.exists():
        return p1

    # relative to PDFDIR by basename
    p2 = PDFDIR / Path(primary_file).name
    if p2.exists():
        return p2

    return None

rows = list(csv.DictReader(open(REG, encoding="utf-8-sig", newline="")))
inc = [r for r in rows if clean(r.get("include_final","")) == "1"]

out_rows = []
have_primary = 0
pdf_found = 0
tei_found = 0

for r in inc:
    primary = clean(r.get("primary_file",""))
    if primary:
        have_primary += 1

    pdfp = resolve_pdf_path(primary)
    pdf_exists = int(pdfp.exists()) if pdfp else 0
    if pdf_exists:
        pdf_found += 1

    teip = None
    tei_exists = 0
    if pdfp:
        teip = TEIDIR / (pdfp.stem + ".tei.xml")
        tei_exists = int(teip.exists())
        if tei_exists:
            tei_found += 1

    out_rows.append({
        "paper_id": clean(r.get("paper_id","")),
        "doi_resolved": clean(r.get("doi_resolved","")),
        "title": clean(r.get("title","")),
        "primary_file": primary,
        "pdf_path_resolved": str(pdfp) if pdfp else "",
        "pdf_exists": pdf_exists,
        "tei_path_from_stem": str(teip) if teip else "",
        "tei_exists": tei_exists,
    })

OUT.parent.mkdir(parents=True, exist_ok=True)
with open(OUT, "w", encoding="utf-8", newline="") as f:
    w = csv.DictWriter(f, fieldnames=list(out_rows[0].keys()))
    w.writeheader()
    w.writerows(out_rows)

print("Included rows:", len(inc))
print("Have primary_file:", have_primary)
print("PDF found:", pdf_found)
print("TEI found via stem:", tei_found)
print("Wrote:", OUT)
